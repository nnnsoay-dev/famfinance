                                {Object.values(debts).filter(item => item.tab === activeTab).map(item => <DebtCard key={item.id} item={item} updateItem={updateItem} onViewDetails={setSelectedDebtId} />)}

                    {selectedDebtId && <DebtDetailModal debt={debts[selectedDebtId]} onClose={() => setSelectedDebtId(null)} onAddRecord={handleAddRecord} onUpdateBalance={updateItem} />}
